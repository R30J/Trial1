Program to Check Character Is Uppercase, Lowercase Alphabet or a Digit or a Special Symbol



                                #include 

                                    int main()
                                    {
                                        char a;
                                        printf("Enter the character:\n");
                                        scanf("%c", &a);
                                    
                                        if (a >= 48 && a <= 57)
                                        {
                                            printf("The character is DIGITVALUE\n");
                                        }
                                        else
                                        {
                                            if (a >= 65 && a <= 90)
                                            {
                                                printf("The character is UPPERCASE\n");
                                            }
                                            else
                                            {
                                                if (a >= 97 && a <= 122)
                                                {
                                                    printf("The character is LOWERCASE\n");
                                                }
                                                else
                                                {
                                                    printf("The character is SPECIAL SYMBOL\n");
                                                }
                                            }
                                        }
                                    }

Find the Greatest Among Three Numbers.



                                #include 

                                    int main()
                                    {
                                        double num1, num2, num3;
                                        printf("Enter three NUMBERS:\n");
                                        scanf("%lf %lf %lf", &num1, &num2, &num3);
                                    
                                        if (num1 > num2 && num1 > num3)
                                        {
                                            printf("%lf is greater than %lf and %lf", num1, num2, num3);
                                        }
                                        else if (num2 > num1 && num2 > num3)
                                        {
                                            printf("%lf is greater than %lf and %lf", num2, num1, num3);
                                        }
                                        else if (num3 > num1 && num3 > num2)
                                        {
                                            printf("%lf is greater than %lf and %lf", num3, num1, num2);
                                        }
                                        else
                                        {
                                            printf("I DON'T KNOW");
                                        }
                                        return 0;
                                    }
